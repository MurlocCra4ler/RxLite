43template <typename T, typename... Us>
44Operator<T, std::tuple<T, Us...>> combineLatest(Observable<Us>... latestObservables) {
45 return [latestObservables...](const Observable<T>& sourceObservable) {
46 return impl::ObservableFactory<std::tuple<T, Us...>>(
47 [sourceObservable, latestObservables...](const Subscriber<std::tuple<T, Us...>>& subscriber) {
48 Subscription subscriptions;
49 auto latestValues = std::make_shared<std::tuple<std::optional<T>, std::optional<Us>...>>();
50 auto completedFlags = std::make_shared<std::array<bool, sizeof...(Us) + 1>>();
51
52 auto emitIfReady = [subscriber = subscriber.shared_from_this(), latestValues]() {
53 if (std::apply([](auto&... values) { return (... && values.has_value()); }, *latestValues)) {
54 subscriber->next(std::apply([](auto&... values) {
55 return std::make_tuple(values.value()...);
56 }, *latestValues));
57 }
58 };
59
60 auto completeIfReady = [subscriber = subscriber.shared_from_this(), completedFlags]() {
61 if (std::apply([](auto... flags) { return (... && flags); }, *completedFlags)) {
62 subscriber->complete();
63 }
64 };
65
66 Observer<T> sourceObserver(
67 [latestValues, emitIfReady](const T& t) {
68 std::get<0>(*latestValues) = t;
69 emitIfReady();
70 },
71 [subscriber = subscriber.shared_from_this()](const std::exception_ptr& err) {
72 subscriber->error(err);
73 },
74 [completedFlags, completeIfReady]() {
75 completedFlags->at(0) = true;
76 completeIfReady();
77 }
78 );
79 subscriptions.add(sourceObservable.subscribe(sourceObserver));
80
81 auto subscribeLatest = [&]<std::size_t... Is>(std::index_sequence<Is...>) {
82 (void)std::initializer_list<int>{
83 (subscriptions.add(
84 latestObservables.subscribe(
85 Observer<Us>(
86 [latestValues, emitIfReady](const Us& value) {
87 std::get<Is + 1>(*latestValues) = value;
88 emitIfReady();
89 },
90 [subscriber = subscriber.shared_from_this()](const std::exception_ptr& err) {
91 subscriber->error(err);
92 },
93 [completedFlags, completeIfReady]() {
94 completedFlags->at(Is + 1) = true;
95 completeIfReady();
96 }
97 )
98 )
99 ), 0)...
100 };
101 };
102
103 subscribeLatest(std::index_sequence_for<Us...>{});
104 return [subscriptions]() mutable {
105 subscriptions.unsubscribe();
106 };
107 }
108 );
109 };

130template <typename T>
131Operator<T, T> distinct() {
132 return [](const Observable<T>& sourceObservable) {
133 return impl::ObservableFactory<T>([sourceObservable](const Subscriber<T>& subscriber) {
134 std::unordered_set<T> seen;
135
136 Observer<T> intermediateObserver(
137 [seen = std::move(seen), subscriber = subscriber.shared_from_this()](const T& t) mutable {
138 auto [_, inserted] = seen.emplace(t);
139 if (inserted) {
140 subscriber->next(t);
141 }
142 },
143 [subscriber = subscriber.shared_from_this()](const std::exception_ptr& err) {
144 subscriber->error(err);
145 },
146 [subscriber = subscriber.shared_from_this()]() { subscriber->complete(); }
147 );
148
149 return [subscription = sourceObservable.subscribe(intermediateObserver)]() mutable {
150 subscription.unsubscribe();
151 };
152 });
153 };

174template <typename T>
175Operator<T, T> distinctUntilChanged() {
176 return [](const Observable<T>& sourceObservable) {
177 return impl::ObservableFactory<T>([sourceObservable](const Subscriber<T>& subscriber) {
178 std::optional<T> lastValue;
179
180 Observer<T> intermediateObserver(
181 [lastValue = std::move(lastValue), subscriber = subscriber.shared_from_this()](const T& t) mutable {
182 if (!lastValue || *lastValue != t) {
183 subscriber->next(t);
184 lastValue = t;
185 }
186 },
187 [subscriber = subscriber.shared_from_this()](const std::exception_ptr& err) {
188 subscriber->error(err);
189 },
190 [subscriber = subscriber.shared_from_this()]() { subscriber->complete(); }
191 );
192

249template <typename T, typename... Ts>
250requires(std::is_same_v<T, Ts> && ...)
251Operator<T, T> merge(const Observable<Ts>&... observables) {
252 return [observables...](const Observable<T>& sourceObservable) {
253 return impl::ObservableFactory<T>([sourceObservable, observables...](const Subscriber<T>& subscriber) {
254 const std::size_t totalSources = sizeof...(Ts) + 1;
255 auto completedCounter = std::make_shared<size_t>(0);
256
257 Observer<T> intermediateObserver(
258 [subscriber = subscriber.shared_from_this()](const T& t) {
259 subscriber->next(t);
260 },
261 [subscriber = subscriber.shared_from_this()](const std::exception_ptr& err) {
262 subscriber->error(err);
263 },
264 [subscriber = subscriber.shared_from_this(), completedCounter, totalSources]() {
265 *completedCounter += 1;
266
267 if (*completedCounter == totalSources) {
268 subscriber->complete();
269 }
270 }
271 );
272
273 Subscription subscription = sourceObservable.subscribe(intermediateObserver);
274 (subscription.add(observables.subscribe(intermediateObserver)), ...);
275
276 return [subscription]() mutable {
277 subscription.unsubscribe();
278 };
279 });
280 };

305template <typename T, typename... Us>
306Operator<T, std::tuple<T, Us...>> withLatestFrom(Observable<Us>... latestObservables) {
307 return [latestObservables...](const Observable<T>& sourceObservable) {
308 return impl::ObservableFactory<std::tuple<T, Us...>>(
309 [sourceObservable, latestObservables...](const Subscriber<std::tuple<T, Us...>>& subscriber) {
310 auto latestValues = std::make_shared<std::tuple<std::optional<Us>...>>();
311
312 // Function to check if all values in the tuple are set
313 auto allSet = [latestValues]() {
314 return std::apply([](auto&... values) {
315 return (... && values.has_value());
316 }, *latestValues);
317 };
318
319 Subscription subscriptions;
320
321 // Subscribe to each latestObservable and update its corresponding value in latestValues
322 auto subscribeLatest = [&]<std::size_t... Is>(std::index_sequence<Is...>) {
323 (void)std::initializer_list<int>{
324 (subscriptions.add(
325 latestObservables.subscribe(
326 Observer<Us>(
327 [latestValues](const Us& value) {
328 std::get<Is>(*latestValues) = value;
329 },
330 [subscriber = subscriber.shared_from_this()](const std::exception_ptr& err) {
331 subscriber->error(err);
332 }
333 )
334 )
335 ), 0)... // Using comma operator to expand the parameter pack
336 };
337 };
338
339 // Create observers dynamically for each latest observable
340 subscribeLatest(std::index_sequence_for<Us...>{});
341
342 // Subscribe to the source observable
343 Observer<T> combinedObserver(
344 [subscriber = subscriber.shared_from_this(), latestValues, allSet](const T& t) {
345 if (allSet()) {
346 subscriber->next(
347 std::tuple_cat(std::make_tuple(t),
348 std::apply([](auto&... values) { return std::make_tuple(values.value()...); }, *latestValues)
349 )
350 );
351 }
352 },
353 [subscriber = subscriber.shared_from_this()](const std::exception_ptr& err) { subscriber->error(err); },
354 [subscriber = subscriber.shared_from_this()]() { subscriber->complete(); }
355 );
356
357 subscriptions.add(sourceObservable.subscribe(combinedObserver));
358 return [subscriptions]() mutable {
359 subscriptions.unsubscribe();
360 };
361 }
362 );
363 };
